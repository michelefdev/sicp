[NOTA CHE IL LOOP E' INFINITO PER QUALSIASI NUMERO NON SIA UN QUADRATO PERFETTO, QUINDI PRIMA SCRIVIAMO UNA PROCEDURA FUNZIONANTE PER I CASI SPECIALI,
POI AGGIUNGIAMO UN "GOOD-ENOUGH" PER COPRIRE ANCHE I QUADRATI NON PERFETTI]


(define (good-enough? guess square)
  (if (< (abs (- (* guess guess) square)) 0.01)
    true
    false
  )
)


(define (avg x y)
  (/ (+ x y)
      2
  )
)

(define (try guess square)
  (if (good-enough? guess square)
      guess
      (try (avg (/ square guess) guess)
           square)))
