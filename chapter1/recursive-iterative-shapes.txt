#Redefine a function my-product that takes two positive integers a and b and returns their product a * b.
(define (my-product a b)
    (define (product-iter a b total)
        (cond
            ( (or (= a 0) (= b 0)) 0)
            ((= b 1) (+ total a))
            (else (product-iter a 
                            (- b 1) 
                            (+ total a)))
        )
    )
    (product-iter a b 0)
)
